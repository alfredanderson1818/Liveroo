⚠️ Firestore: sin permisos para leer algunas colecciones. Agrega las reglas abajo en Firebase Console → Firestore → Rules
🔐 Reglas de Firestore necesarias
Copia esto en Firebase Console → Firestore → Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Admin tiene acceso total
    function isAdmin() {
      return request.auth != null &&
        request.auth.uid in [
          'REEMPLAZA_CON_TU_UID_ADMIN'
        ];
    }

    // Acceso admin total
    match /{document=**} {
      allow read, write: if isAdmin();
    }

    // Usuarios: lectura pública, escritura propia
    match /users/{uid} {
      allow read: if true;
      allow write: if request.auth.uid == uid || isAdmin();
    }

    // Órdenes: solo comprador, vendedor o admin
    match /orders/{id} {
      allow read: if request.auth != null &&
        (request.auth.uid == resource.data.buyerId ||
         request.auth.uid == resource.data.sellerId ||
         isAdmin());
      allow write: if request.auth != null || isAdmin();
    }

    // Depósitos: solo el usuario dueño o admin
    match /deposit_requests/{id} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == resource.data.uid || isAdmin());
    }

    // Wallet txs
    match /wallet_txs/{id} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == resource.data.uid || isAdmin());
    }

    // Productos y tiendas: lectura pública
    match /products/{id} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    match /stores/{id} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Streams
    match /streams/{id} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
